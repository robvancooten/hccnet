const points = [
    new THREE.Vector3(0, 0, 0),
    new THREE.Vector3(5, 2, -3),
    new THREE.Vector3(10, 0, 0),
    new THREE.Vector3(15, 3, 5)
];

const curve = new THREE.CatmullRomCurve3(points, false, "catmullrom", 0.5);


const objectsOnCurve = [];
const loader = new THREE.GLTFLoader();

function loadObject(path, offsetT) {
    loader.load(path, (gltf) => {
        const model = gltf.scene;
        scene.add(model);

        objectsOnCurve.push({
            mesh: model,
            t: offsetT   // startpositie op de curve
        });
    });
}

loadObject("car.glb", 0.0);
loadObject("car.glb", 0.33);
loadObject("car.glb", 0.66);

function animate(time) {
    requestAnimationFrame(animate);

    const speed = 0.0003;

    objectsOnCurve.forEach((obj) => {
        obj.t += speed;
        if (obj.t > 1) obj.t -= 1;

        const pos = curve.getPointAt(obj.t);
        const tangent = curve.getTangentAt(obj.t);

        obj.mesh.position.copy(pos);

        // Rotatie mee met de curve
        const up = new THREE.Vector3(0, 1, 0);
        const quaternion = new THREE.Quaternion().setFromUnitVectors(up, tangent.clone().normalize());
        obj.mesh.quaternion.copy(quaternion);
    });

    renderer.render(scene, camera);
}

animate();
